
\usepackage[a4paper, hmargin=25mm, vmargin=30mm, top=20mm]{geometry} % Use A4 paper and set margins
\usepackage{hyperref} % Colors for links, text and headings

% Font settings
\usepackage[proportional,scaled=1]{erewhon}
\usepackage{color} % Required for custom colors
\definecolor{slateblue}{rgb}{0.17,0.22,0.34}

% Section Header Styling
\usepackage{sectsty} % Allows customization of titles
\sectionfont{\color{slateblue}} % Color section titles
\setcounter{secnumdepth}{0} % Suppress section numbering

% Prevent Paragraph Indentation
\setlength\parindent{0pt} % Stop paragraph indentation

% Text width for tabbing environments
\newlength{\smallertextwidth}
\setlength{\smallertextwidth}{\textwidth}
\addtolength{\smallertextwidth}{-2cm}

% Define square bullet
\newcommand{\sqbullet}{~\vrule height 1ex width .8ex depth -.2ex} % Custom square bullet point definition

% Customize Title
\renewcommand{\title}[1]{
	\huge{\color{slateblue}\textbf{#1}}\\ % Header section name and color
	\rule{\textwidth}{0.5mm}\\ % Rule under the header
}

% Tabbing environment with no vertical space padding
\newenvironment{nstabbing}
{\setlength{\topsep}{0pt}
	\setlength{\partopsep}{0pt}
	\tabbing}
{\endtabbing}

% Generic Tabbed Text Command
\newcommand{\tabbedblock}[1]{
	\begin{tabbing}
		\hspace{2cm} \= \kill
		#1
	\end{tabbing}
}

% Academic Job Command
\newcommand{\jobMultiLine}[5]{
	\begin{tabbing}
	\hspace{2cm} \= \kill
	\textbf{#1} \> #3 \\
	\textbf{#2} \>\+ \textit{#4} \\
	\end{tabbing}
}

% Small header
\newcommand{\smallHeader}[1]{
	\begin{tabbing}
		\hspace{2cm} \= \kill
		\> \sc{#1}
	\end{tabbing}
}

% Achievement Command
\newcommand{\achievement}[2]{
	\begin{nstabbing}
		\hspace{2cm} \= \kill
		\textbf{#1} \> #2
	\end{nstabbing}
}

% Project/Teaching Command
\newcommand{\project}[5]{
	\begin{tabbing}
		\hspace{2cm} \= \kill
		\textbf{#1} \> #3 \\
		\textbf{#2} \> #4 \\
		\> 
		\begin{minipage}{\smallertextwidth}
			\vspace{2mm}
			#5
		\end{minipage}
	\end{tabbing}
	\vspace{2mm}
}

% Service
\newcommand{\positionBlock}[3]{
	\parbox[t]{2cm}{
	\textbf{#1}\\
	\textbf{#2}
	}
	\hfill
	\parbox[t]{\smallertextwidth}{
		#3
	}
	\vspace{2mm}
}

% Skill Command
\newcommand{\skillgroup}[2]{
	\begin{tabbing}
		\hspace{5mm} \= \kill
		\sqbullet \>\+ \textbf{#1} \\
		\begin{minipage}{\smallertextwidth}
			\vspace{2mm}
			#2
		\end{minipage}
	\end{tabbing}
}

% Bullet Point Command
\newcommand{\bulletPoint}[1]{
	\begin{nstabbing}
		\hspace{5mm} \= \kill
		\sqbullet \> #1
	\end{nstabbing}
}

% Reference Command
\newcommand{\reference}[4]{
	\parbox{0.5\textwidth}{
		\begin{tabbing}
			\hspace{2.75cm} \= \hspace{4cm} \= \kill % Spacing within the block
			{\bf Name:} \> #1 \\ % Referee name
			{\bf Institution:} \> #2 \\ % Referee company
			{\bf Position:} \> #3 \\ % Referee job title 
			{\bf Email:} \> #4
		\end{tabbing}
	}
}

% Piece for Worklist
\newcommand{\piece}[4]{
	\begin{nstabbing}
		\hspace{2cm} \= \kill
		\textbf{#1} \> \textit{#2} for #3, #4'
	\end{nstabbing}
}
